package Logica;

/**
 *
 * @author usuario
 */
public enum Estado {
    REGISTRADA,
    CANCELADA,
    PAGADA,
    FACTURADA;

    static Estado obtenerEstado(String estado){
        if(estado.toUpperCase().equals("REGISTRADA"))
            return REGISTRADA;
        else if(estado.toUpperCase().equals("CANCELADA"))
            return CANCELADA;
        else if(estado.toUpperCase().equals("PAGADA"))
            return PAGADA;
        else 
            return FACTURADA;
    }
}
