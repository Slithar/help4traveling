    public boolean correoValido(){
        int cantArrobas = 0;
        for(int i = 0; i < this.email.length(); i++){
            if(this.email.charAt(i) == '@')
                cantArrobas++;
        }
        if(cantArrobas == 1)
            return true;
        else
            return false;
    }
